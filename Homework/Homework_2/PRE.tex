		\subsection{Part I}

		Since $m=n=2$, the possible ranks are $1,2,3,4$, and $T_1$ is the sum of the two

		There are $\binom{4}{2}=6$ equally likely ways to choose the two ranks for the
		first sample:
		\[
		\{1,2\},\ \{1,3\},\ \{1,4\},\ \{2,3\},\ \{2,4\},\ \{3,4\}.
		\]

		\[
		3,\ 4,\ 5,\ 5,\ 6,\ 7.
		\]
		
		Hence the sampling distribution of $T_1$ under $H_0$ is
		
		\[
		\begin{array}{c|ccccc}
			t & 3 & 4 & 5 & 6 & 7\\
			\hline
			\mathbb{P}(T_1=t) & \frac16 & \frac16 & \frac13 & \frac16 & \frac16
		\end{array}
		\]
		
		Next,
		$E[T_1] = \sum t\,\mathbb{P}(T_1=t)
		= 3\cdot\frac16 + 4\cdot\frac16 + 5\cdot\frac13 + 6\cdot\frac16 + 7\cdot\frac16
		= \frac{30}{6} = 5$.
		
		To compute the variance, first calculate
		$E[T_1^2] =
		3^2\cdot\frac16 + 4^2\cdot\frac16 + 5^2\cdot\frac13 + 6^2\cdot\frac16 + 7^2\cdot\frac16
		= \frac{80}{3}$.
		
		Therefore
		$\operatorname{Var}(T_1)
		= E[T_1^2] - (E[T_1])^2
		= \frac{80}{3} - 25
		= \frac{5}{3}$.
		
		Thus,
		\[
		E[T_1] = 5,  \operatorname{Var}(T_1) = \frac{5}{3}.
		\]

		\subsection{Part III}
		Under $H_0$, the combined sample of size $m+n$ comes from the same continuous distribution, so the ranks $1,2,\dots,m+n$ are randomly assigned to the observations. Let $R_i$ denote the rank of $X_i$. Then $T_1=\sum_{i=1}^m R_i$.
		
		\medskip
		\noindent\emph{Expected Value.}
		By the hint, each $R_i$ is equally likely to be any value in $\{1,2,\dots,m+n\}$, so $E[R_i]=\frac{1+2+\cdots+(m+n)}{m+n}=\frac{m+n+1}{2}$. Therefore, by linearity of expectation,
		$$E[T_1]=E\!\left[\sum_{i=1}^m R_i\right]=\sum_{i=1}^m E[R_i]=m\cdot\frac{m+n+1}{2}=\frac{m(m+n+1)}{2}.$$
		
		\medskip
		\noindent\emph{Variance.}
		Write $N=m+n$. Then
		$\operatorname{Var}(T_1)=\sum_{i=1}^m \operatorname{Var}(R_i)+2\sum_{1\le i<j\le m}\operatorname{Cov}(R_i,R_j).$ Since $R_i$ is uniform on $\{1,\dots,N\}$, we have $E[R_i]=\frac{N+1}{2}$ and $E[R_i^2]=\frac{1^2+2^2+\cdots+N^2}{N}=\frac{(N+1)(2N+1)}{6}$, so
		$$\operatorname{Var}(R_i)=E[R_i^2]-(E[R_i])^2=\frac{(N+1)(2N+1)}{6}-\frac{(N+1)^2}{4}=\frac{N^2-1}{12}.$$
		
		For $i\neq j$, $(R_i,R_j)$ is a uniformly random ordered pair of distinct values from $\{1,\dots,N\}$, so there are $N(N-1)$ equally likely outcomes. Using the identity
		$$\sum_{\substack{a,b=1\\a\neq b}}^{N}ab=\left(\sum_{a=1}^N a\right)^{\!2}-\sum_{a=1}^N a^2=\frac{N^2(N+1)^2}{4}-\frac{N(N+1)(2N+1)}{6},$$
		we compute
		\begin{align*}
			E[R_iR_j]
			&=\frac{1}{N(N-1)}\left[\frac{N^2(N+1)^2}{4}-\frac{N(N+1)(2N+1)}{6}\right]
			=\frac{N(N+1)}{N(N-1)}\left[\frac{N(N+1)}{4}-\frac{2N+1}{6}\right]\\
			&=\frac{N+1}{N-1}\cdot\frac{3N(N+1)-2(2N+1)}{12}
			=\frac{N+1}{N-1}\cdot\frac{3N^2-N-2}{12}
			=\frac{N+1}{N-1}\cdot\frac{(N-1)(3N+2)}{12}
			=\frac{(N+1)(3N+2)}{12}.
		\end{align*}
		Hence
		\begin{align*}
			\operatorname{Cov}(R_i,R_j)
			&=E[R_iR_j]-E[R_i]\,E[R_j]
			=\frac{(N+1)(3N+2)}{12}-\left(\frac{N+1}{2}\right)^{\!2}
			=(N+1)\left[\frac{3N+2}{12}-\frac{N+1}{4}\right]
			=(N+1)\cdot\frac{-1}{12}
		\end{align*} =$-\frac{N+1}{12}.$ Then substituting back and using $2\binom{m}{2}=m(m-1)$, we have
		$\operatorname{Var}(T_1)=m\cdot\frac{N^2-1}{12}+m(m-1)\!\left(-\frac{N+1}{12}\right)=\frac{m(N^2-1)-m(m-1)(N+1)}{12}=\frac{m(N+1)\bigl[(N-1)-(m-1)\bigr]}{12}=\frac{m(N+1)(N-m)}{12}.$
		
		Since $N=m+n$ gives $N-m=n$, we conclude that
		$$E[T_1]=\frac{m(m+n+1)}{2},  \operatorname{Var}(T_1)=\frac{mn(m+n+1)}{12}$$

		\subsection{Part I}
		
		We test
		$H_0:\mu_1-\mu_2=\Delta_0$
		versus
		$H_A:\mu_1-\mu_2>\Delta_0$,
		assuming both populations are normal with common variance $\sigma^2$. Let $\bar X$ and $\bar Y$ denote the sample means from samples of sizes
		$m$ and $n$, and define the pooled variance estimator
		$S_p^2=\frac{(m-1)S_1^2+(n-1)S_2^2}{m+n-2}$. The pooled two-sample $t$ statistic is
		$T=\frac{(\bar X-\bar Y)-\Delta_0}{S_p\sqrt{\frac1m+\frac1n}}$. Under $H_0$, $T\sim t_\nu$ with $\nu=m+n-2$ degrees of freedom.  
		Thus the rejection region for a level $\alpha$ upper-tailed test is
		$T>t_{\alpha,\nu}$ where $P(t_\nu>t_{\alpha,\nu})=\alpha$.
		
		Under a true difference $\mu_1-\mu_2=\Delta$, the statistic has a
		noncentral $t$ distribution with noncentrality parameter $\delta=\frac{(\mu_1-\mu_2)-\Delta_0}{\sigma\sqrt{\frac1m+\frac1n}}
		=\frac{\Delta-\Delta_0}{\sigma\sqrt{\frac1m+\frac1n}}$.
		
		Thus $T\sim t_\nu(\delta)$. The power function is therefore $\pi(\Delta)
		=P_\Delta(T>t_{\alpha,\nu})
		=1-F_{t_\nu(\delta)}(t_{\alpha,\nu})$, where $F_{t_\nu(\delta)}$ is the CDF of the noncentral $t$ distribution.

		\section{Question 5}
		{\color{deepred}\fontspec{Georgia} Two-Sample Pooled Test and Likelihood Ratio}
		
		Recall the pooled two-sample t-procedure, which assumes the model:
		
		\medskip
		
		\underline{Assumptions}
		
		\begin{enumerate}
			\item $X_1, X_2, \ldots, X_m$ is a random sample from a normal distribution with mean $\mu_1$ and variance $\sigma^2$. The variance $\sigma^2$ is unknown.
			
			\item $Y_1, Y_2, \ldots, Y_n$ is a random sample from a normal distribution with mean $\mu_2$ and variance $\sigma^2$. The variance $\sigma^2$ is unknown.
			
			\item The $X$ and $Y$ samples are independent of one another.
		\end{enumerate}
		
		\medskip
		
		Consider the two tailed hypothesis
		\[
		H_0 : \mu_1 - \mu_2 = 0 
		\text{versus} 
		H_A : \mu_1 - \mu_2 \neq 0 .
		\]
		
		\medskip
		
		\emph{Perform the following tasks}
		
		\begin{enumerate}
			
			\item[i] Derive an expression for the likelihood
			\[
			\mathcal{L}(\theta) = \mathcal{L}(\mu_1,\mu_2,\sigma^2).
			\]
			
			\item[ii] Find the maximum likelihood estimator of
			\[
			\theta = (\mu_1,\mu_2,\sigma^2).
			\]
			
			Note that you are not required to perform the \emph{second derivative test} for this problem.
			
			\item[iii] Derive the likelihood ratio test statistic $\lambda$. Note that you are not required to perform the \emph{second derivative test} for this problem.
			
			\item[iv] Show that the likelihood ratio test can be expressed in terms of the t-statistic $(T)$, i.e., there exists a function $f$ such that
			\[
			\lambda = f(T).
			\]
			
		\end{enumerate}
		
		\textbf{Answer}
		
		\subsection{Part I}
		
		Let $x_1,\dots,x_m$ be the observed $X$-sample and $y_1,\dots,y_n$ the observed $Y$-sample. Under the model,
		$
		X_i \stackrel{iid}{\sim} N(\mu_1,\sigma^2), 
		Y_j \stackrel{iid}{\sim} N(\mu_2,\sigma^2),
		$
		with the two samples independent.
		
		Define
		$
		\bar X=\frac1m\sum_{i=1}^m X_i, 
		\bar Y=\frac1n\sum_{j=1}^n Y_j,
		$
		and let
		$
		SSE_1=\sum_{i=1}^m (X_i-\bar X)^2, 
		SSE_2=\sum_{j=1}^n (Y_j-\bar Y)^2.
		$
		
		
		By independence,
		$
		\mathcal L(\mu_1,\mu_2,\sigma^2)
		=
		\prod_{i=1}^m \frac{1}{\sqrt{2\pi\sigma^2}}
		\exp\!\left(-\frac{(x_i-\mu_1)^2}{2\sigma^2}\right)
		\prod_{j=1}^n \frac{1}{\sqrt{2\pi\sigma^2}}
		\exp\!\left(-\frac{(y_j-\mu_2)^2}{2\sigma^2}\right).
		$
		Thus
		$
		\mathcal L(\mu_1,\mu_2,\sigma^2)
		=
		(2\pi\sigma^2)^{-(m+n)/2}
		\exp\!\left(
		-\frac{1}{2\sigma^2}
		\left[
		\sum_{i=1}^m (x_i-\mu_1)^2+\sum_{j=1}^n (y_j-\mu_2)^2
		\right]
		\right).
		$
		Equivalently, the log-likelihood is
		$
		\ell(\mu_1,\mu_2,\sigma^2)
		=
		-\frac{m+n}{2}\log(2\pi)
		-\frac{m+n}{2}\log(\sigma^2)
		-\frac{1}{2\sigma^2}
		\left[
		\sum_{i=1}^m (x_i-\mu_1)^2+\sum_{j=1}^n (y_j-\mu_2)^2
		\right].
		$
		
		\clearpage
		
		\subsection{Part II}
		
		To maximize $\mathcal L$, equivalently maximize $\ell$.
		
		Differentiating with respect to $\mu_1$ and $\mu_2$ gives
		$
		\hat\mu_1=\bar X,  \hat\mu_2=\bar Y.
		$
		
		Substituting these into the log-likelihood, we maximize in $\sigma^2$:
		$
		\ell(\bar X,\bar Y,\sigma^2)
		=
		-\frac{m+n}{2}\log(2\pi)
		-\frac{m+n}{2}\log(\sigma^2)
		-\frac{1}{2\sigma^2}
		\left[
		\sum_{i=1}^m (x_i-\bar X)^2+\sum_{j=1}^n (y_j-\bar Y)^2
		\right].
		$
		Hence
		$
		\hat\sigma^2
		=
		\frac{1}{m+n}
		\left[
		\sum_{i=1}^m (x_i-\bar X)^2+\sum_{j=1}^n (y_j-\bar Y)^2
		\right]
		=
		\frac{SSE_1+SSE_2}{m+n}.
		$
		
		Therefore the unrestricted MLE is
		$
		\hat\theta=
		\left(
		\bar X,\bar Y,\frac{SSE_1+SSE_2}{m+n}
		\right).
		$
		
		\clearpage
		
		\subsection{Part III}
		
		Under $H_0:\mu_1-\mu_2=0$, we write $\mu_1=\mu_2=\mu$. Then the likelihood becomes
		$
		\mathcal L_0(\mu,\sigma^2)
		=
		(2\pi\sigma^2)^{-(m+n)/2}
		\exp\!\left(
		-\frac{1}{2\sigma^2}
		\left[
		\sum_{i=1}^m (x_i-\mu)^2+\sum_{j=1}^n (y_j-\mu)^2
		\right]
		\right).
		$
		
		The MLE of $\mu$ under $H_0$ is the combined sample mean
		$
		\hat\mu_0=\bar Z=\frac{m\bar X+n\bar Y}{m+n}.
		$
		
		The constrained MLE of $\sigma^2$ is
		$
		\hat\sigma_0^2
		=
		\frac{1}{m+n}
		\left[
		\sum_{i=1}^m (x_i-\bar Z)^2+\sum_{j=1}^n (y_j-\bar Z)^2
		\right].
		$
		
		Using the identity
		$
		\sum_{i=1}^m (x_i-\bar Z)^2+\sum_{j=1}^n (y_j-\bar Z)^2
		=
		SSE_1+SSE_2+\frac{mn}{m+n}(\bar X-\bar Y)^2,
		$
		we get
		$
		\hat\sigma_0^2
		=
		\frac{1}{m+n}
		\left[
		SSE_1+SSE_2+\frac{mn}{m+n}(\bar X-\bar Y)^2
		\right].
		$
		
		The likelihood ratio statistic is
		$
		\lambda
		=
		\frac{\sup_{H_0}\mathcal L}{\sup_{\Theta}\mathcal L}.
		$
		Since at the maximizers the exponential term becomes $e^{-(m+n)/2}$ in both numerator and denominator, we obtain
		$
		\lambda
		=
		\left(\frac{\hat\sigma^2}{\hat\sigma_0^2}\right)^{(m+n)/2}.
		$
		Thus
		$
		\lambda
		=
		\left(
		\frac{SSE_1+SSE_2}
		{SSE_1+SSE_2+\frac{mn}{m+n}(\bar X-\bar Y)^2}
		\right)^{(m+n)/2}.
		$
		
		\clearpage
		
		\subsection{Part iv}
		
		Recall the pooled variance
		$
		S_p^2=\frac{SSE_1+SSE_2}{m+n-2},
		$
		and the pooled two-sample $t$-statistic
		$
		T
		=
		\frac{\bar X-\bar Y}
		{S_p\sqrt{\frac1m+\frac1n}}.
		$
		Then
		$
		T^2
		=
		\frac{(\bar X-\bar Y)^2}
		{S_p^2\left(\frac1m+\frac1n\right)}
		=
		\frac{(\bar X-\bar Y)^2}
		{\frac{SSE_1+SSE_2}{m+n-2}\cdot \frac{m+n}{mn}}
		=
		\frac{mn}{m+n}\cdot \frac{(m+n-2)(\bar X-\bar Y)^2}{SSE_1+SSE_2}.
		$
		Hence
		$
		\frac{mn}{m+n}(\bar X-\bar Y)^2
		=
		\frac{SSE_1+SSE_2}{m+n-2}\,T^2.
		$
		
		Substitute this into the formula for $\lambda$:
		$
		\lambda
		=
		\left(
		\frac{SSE_1+SSE_2}
		{SSE_1+SSE_2+\frac{SSE_1+SSE_2}{m+n-2}T^2}
		\right)^{(m+n)/2}.
		$
		Factor out $SSE_1+SSE_2$ from the denominator:
		$
		\lambda
		=
		\left(
		\frac{1}
		{1+\frac{T^2}{m+n-2}}
		\right)^{(m+n)/2}.
		$
		
		Therefore $\lambda=f(T)$ with
		$
		f(t)=\left(1+\frac{t^2}{m+n-2}\right)^{-(m+n)/2}.
		$
		
		So the likelihood ratio test is an equivalent test to the pooled two-sample $t$-test, since $\lambda$ is a monotone decreasing function of $|T|$.

		\section{Question 6}
		
		{\color{deepred}\fontspec{Georgia} Parametric and Non-parametric Paired Procedures}
		
		\medskip
		
		Consider testing the stopping distance [ft] of two types of tires on 6 different cars. 
		The response variable is the stopping distance of the car after the brakes are applied 
		at 60 mph. The two types of tires are: Winter and High Performance. 
		Does the data set indicate that the stopping distance differs between the two tire types?
		
		\medskip
		
		\begin{center}
			\begin{tabular}{ccc}
				\hline
				Car & High Performance & Winter \\
				\hline
				1 & 226.6 & 239.5 \\
				2 & 230.5 & 257.5 \\
				3 & 220.0 & 236.8 \\
				4 & 215.9 & 223.9 \\
				5 & 231.5 & 248.0 \\
				6 & 235.5 & 253.6 \\
				\hline
			\end{tabular}
		\end{center}
		
		\medskip
		
		\emph{Perform the following tasks}
		
		\begin{enumerate}
			
			\item[i] Assuming the stopping distance for both tire types are approximately normally distributed, run the paired two-sample t-procedure to test the claim
			
			\[
			H_0 : \mu_D = 0 
			\text{versus} 
			H_A : \mu_D \neq 0 .
			\]
			
			Make sure to include all relevant steps in the testing procedure for full credit.
			
			\item[ii] Assuming the stopping distance for $D = X - Y$ is symmetric, run the Wilcoxon signed-rank procedure to test the claim
			
			\[
			H_0 : \mu_D = 0 
			\text{versus} 
			H_A : \mu_D \neq 0 .
			\]
			
			Make sure to include all relevant steps in the testing procedure for full credit.
			
		\end{enumerate}
		
		
		\textbf{Answer}
		
		Let $X$ denote the stopping distance for High Performance tires and $Y$ the stopping distance for Winter tires. Define paired differences
		$
		D_i = X_i - Y_i,  i=1,\dots,6.
		$
		
		From the data we obtain
		$
		D = (-12.9,-27.0,-16.8,-8.0,-16.5,-18.1).
		$
		
		
		\subsection{Part I}
		
		We test
		$
		H_0:\mu_D=0
		\text{vs} 
		H_A:\mu_D\ne 0.
		$
		
		First compute the sample mean
		$
		\bar D=\frac{1}{6}\sum_{i=1}^6 D_i
		= -16.55.
		$
		
		Next compute the sample standard deviation
		$
		s_D \approx 6.49.
		$
		
		The test statistic is
		$
		t=\frac{\bar D-0}{s_D/\sqrt{n}}
		=\frac{-16.55}{6.49/\sqrt{6}}
		\approx -6.25.
		$
		
		The degrees of freedom are
		$
		df=n-1=5.
		$
		
		For a two–sided test, the critical value at $\alpha=0.05$ is approximately
		$
		t_{0.025,5}\approx 2.571.
		$
		
		Since $|t|=6.25>2.571$, we reject $H_0$. The $p$–value is very small ($p<0.01$).
		
		\medskip
		
		There is strong evidence that the mean stopping distance differs between the two tire types. Because $\bar D<0$, High Performance tires tend to have shorter stopping distances than Winter tires.
		
		\clearpage
		
		\subsection{Part II}
		
		Again test
		$
		H_0:\mu_D=0
		\text{vs} 
		H_A:\mu_D\ne 0.
		$
		
		Compute the absolute differences
		$
		|D|=(12.9,27.0,16.8,8.0,16.5,18.1).
		$
		
		Rank them from smallest to largest:
		
		\begin{center}
			\begin{tabular}{ccc}
				\hline
				$D_i$ & $|D_i|$ & Rank \\
				\hline
				$-8.0$ & 8.0 & 1 \\
				$-12.9$ & 12.9 & 2 \\
				$-16.5$ & 16.5 & 3 \\
				$-16.8$ & 16.8 & 4 \\
				$-18.1$ & 18.1 & 5 \\
				$-27.0$ & 27.0 & 6 \\
				\hline
			\end{tabular}
		\end{center}
		
		All differences are negative, so
		
		$
		W_+=0,  W_- = 1+2+3+4+5+6=21.
		$
		
		The Wilcoxon statistic is
		$
		W=\min(W_+,W_-)=0.
		$
		
		For $n=6$, the exact two–sided $p$–value is
		$
		p=\frac{2}{2^6}=0.03125.
		$
		
		
		Since $p<0.05$, we reject $H_0$. The Wilcoxon signed–rank test also indicates that the stopping distance differs between the tire types.
		
		Overall, both the paired $t$–test and the Wilcoxon signed–rank test provide evidence that the stopping distances differ between High Performance and Winter tires, with High Performance tires producing shorter stopping distances on average.

		\section{Question 7}
		
		{\color{deepred}\fontspec{Georgia} Categorical Hypothesis Testing}
		
		\medskip
		
		Binomial data gathered from more than one population are often presented in a contingency table. 
		For the case of two populations, the table might look like
		
		\begin{center}
			\begin{tabular}{c|c|c|c}
				& Population 1 & Population 2 & Total \\
				\hline
				Successes & $S_1$ & $S_2$ & $S = S_1 + S_2$ \\
				Failures & $F_1$ & $F_2$ & $F = F_1 + F_2$ \\
				\hline
				Total & $m$ & $n$ & $N = m + n$
			\end{tabular}
		\end{center}
		
		where Population 1 is $\text{Binomial}(m,p_1)$, with $S_1$ successes and $F_1$ failures, and Population 2 is $\text{Binomial}(n,p_2)$, with $S_2$ successes and $F_2$ failures. 
		We will motivate both the two-sample $z$-test for proportions and the chi-squared test for association based on the $2\times2$ contingency table. 
		The hypothesis of interest is:
		
		\[
		H_0: p_1 = p_2
		\text{versus} 
		H_A: p_1 \neq p_2
		\]
		
		\emph{Perform the following tasks}
		
		\begin{enumerate}
			
			\item[i] Show that a test can be based on the statistic
			\[
			W = \frac{(\hat p_1 - \hat p_2)^2}{\left(\frac{1}{m} + \frac{1}{n}\right)\left(\hat p (1-\hat p)\right)},
			\]
			where $\hat p_1 = S_1/m$, $\hat p_2 = S_2/n$, and
			\[
			\hat p = \frac{S_1 + S_2}{m+n}.
			\]
			Also show that as $m,n \to \infty$, the distribution of $W$ approaches $\chi^2_1$.
			
			\item[ii] Another way of measuring the departure from $H_0$ is by calculating the expected frequency table. 
			This table is constructed by conditioning on the marginal totals and filling in the table according to $H_0: p_1=p_2$, that is,
			
			\medskip
			
			\begin{center}
				Expected Frequencies
				
				\begin{tabular}{c|c|c|c}
					& Population 1 & Population 2 & Total \\
					\hline
					Successes & $\frac{mS}{m+n}$ & $\frac{nS}{m+n}$ & $S = S_1 + S_2$ \\
					Failures & $\frac{mF}{m+n}$ & $\frac{nF}{m+n}$ & $F = F_1 + F_2$ \\
					\hline
					Total & $m$ & $n$ & $N = m+n$
				\end{tabular}
			\end{center}
			
			Using the expected frequency table, a statistic $W^*$ is computed by going through the cells of the tables and computing
			
			\[
			W^* =
			\sum \frac{(\text{observed} - \text{expected})^2}{\text{expected}}
			\]
			
			\[
			=
			\frac{\left(S_1 - \frac{mS}{m+n}\right)^2}{\frac{mS}{m+n}}
			+
			\frac{\left(S_2 - \frac{nS}{m+n}\right)^2}{\frac{nS}{m+n}}
			+
			\frac{\left(F_1 - \frac{mF}{m+n}\right)^2}{\frac{mF}{m+n}}
			+
			\frac{\left(F_2 - \frac{nF}{m+n}\right)^2}{\frac{nF}{m+n}}.
			\]
			
			Show, algebraically, that $W = W^*$ and hence that $W^*$ is asymptotically chi-squared.
			
			\item[iii] A famous medical experiment was conducted by Joseph Lister in the late 1800s. 
			Mortality associated with surgery was quite high, and Lister conjectured that the use of disinfectant, carbolic acid, would help. 
			Over a period of several years Lister performed 75 amputations with and without carbonic acid. 
			The data are given here:
			
			\begin{center}
				Carbolic acid Used?
				
				\begin{tabular}{c|c|c}
					& Yes & No \\
					\hline
					Patient lived & 34 & 19 \\
					Patient died & 6 & 16
				\end{tabular}
			\end{center}
			
			Use these data to test whether the use of carbolic acid is associated with patient mortality.
			
		\end{enumerate}
		
		
		\textbf{Answer}
		
		
		\subsection{Part I}
		Under $H_0:p_1=p_2=:p$, we have $\hat p_1=S_1/m$ and $\hat p_2=S_2/n$ with
		$\mathbb E(\hat p_1-\hat p_2)=0$ and
		$\mathrm{Var}(\hat p_1-\hat p_2)=p(1-p)\left(\frac1m+\frac1n\right)$.
		By the central limit theorem,
		$
		\frac{\hat p_1-\hat p_2}
		{\sqrt{p(1-p)\left(\frac1m+\frac1n\right)}}
		\Rightarrow N(0,1)
		$
		as $m,n\to\infty$.
		Replacing $p$ by the pooled estimator
		$
		\hat p=\frac{S_1+S_2}{m+n},
		$
		the large-sample statistic becomes
		$
		Z=\frac{\hat p_1-\hat p_2}
		{\sqrt{\hat p(1-\hat p)\left(\frac1m+\frac1n\right)}}.
		$
		Hence
		$
		W=Z^2=\frac{(\hat p_1-\hat p_2)^2}
		{\left(\frac1m+\frac1n\right)\hat p(1-\hat p)}.
		$
		Since $Z\Rightarrow N(0,1)$, it follows that $W\Rightarrow\chi^2_1$.
		
		\subsection{Part II}
		
		Define $N=m+n$, $S=S_1+S_2$, and $F=F_1+F_2$.
		Let $A=S_1-\frac{mS}{N}$. Then
		$
		S_2-\frac{nS}{N}=-A,
		$
		$
		F_1-\frac{mF}{N}=-A,
		$
		$
		F_2-\frac{nF}{N}=A.
		$
		Hence
		$
		W^*=A^2\!\left(\frac{N}{mS}+\frac{N}{nS}+\frac{N}{mF}+\frac{N}{nF}\right)
		= A^2N\!\left(\frac1m+\frac1n\right)\!\left(\frac1S+\frac1F\right).
		$
		Since $\frac1S+\frac1F=\frac{N}{SF}$,
		$
		W^*=A^2\frac{N^2}{SF}\left(\frac1m+\frac1n\right).
		$
		Now
		$
		A=S_1-\frac{mS}{N}=\frac{mS_2-nS_1}{N},
		$
		so
		$
		A^2=\frac{(mS_2-nS_1)^2}{N^2}.
		$
		Therefore
		$
		W^*=\frac{(mS_2-nS_1)^2}{SF}\left(\frac1m+\frac1n\right).
		$
		Also
		$
		\hat p_1-\hat p_2=\frac{S_1}{m}-\frac{S_2}{n}
		=\frac{nS_1-mS_2}{mn}.
		$
		Since $\hat p(1-\hat p)=\frac{SF}{N^2}$ and
		$\frac1m+\frac1n=\frac{N}{mn}$, substituting into
		$
		W=\frac{(\hat p_1-\hat p_2)^2}
		{\left(\frac1m+\frac1n\right)\hat p(1-\hat p)}
		$
		gives
		$
		W=\frac{(mS_2-nS_1)^2}{SF}\left(\frac1m+\frac1n\right)=W^*.
		$
		Thus $W^*$ is also asymptotically $\chi^2_1$.
		
		\subsection{Part III}
		From the table we take “success’’ to mean death, giving
		$m=40$, $n=35$, $S_1=6$, $S_2=16$.
		Then $\hat p_1=6/40=0.15$, $\hat p_2=16/35\approx0.4571$, and
		$\hat p=22/75\approx0.2933$.
		The statistic is
		$
		W=\frac{(\hat p_1-\hat p_2)^2}
		{\left(\frac1{40}+\frac1{35}\right)\hat p(1-\hat p)}
		\approx 8.49.
		$
		Under $H_0$, $W\sim\chi^2_1$ asymptotically, giving
		$p\approx0.0036$.
		We therefore reject $H_0$ and conclude that use of carbolic acid is
		associated with mortality; the mortality rate is about $15\%$ with
		carbolic acid versus $45.7\%$ without it.
		
		
		\clearpage
		
		
		\section{Question 8}
		
		{\color{deepred}\fontspec{Georgia} Goodness-of-fit}
		
		\medskip
		
		Construct a Chi-squared goodness-of-fit test to assess whether or not a dataset $(X)$ is normally distributed, i.e., test the null alternative pair:
		
		\[
		H_0 : X \sim \text{Normal}
		\text{versus} 
		H_A : X \text{ is not Normal}.
		\]
		
		To accomplish this task, you must first convert the continuous observed data into a frequency table, i.e., the heights of the bars on a histogram. 
		Then apply the goodness-of-fit test based on the observed and expected counts assuming the null hypothesis $H_0: X\sim\text{Normal}$ is true.
		
		Some considerations follow:
		
		\begin{itemize}
			
			\item Let $K$ be the number of \emph{bins} in the histogram. Note that changing the number of bins will consequently change the degrees of freedom of the chi-squared distribution.
			
			\item Don't forget that you have to estimate both $\mu$ and $\sigma^2$ in order to compute the expected counts, which also impacts the degrees of freedom of the chi-squared distribution.
			
			\item If you are successful in solving this problem, maybe simulate a few normal and non-normal datasets and compare the qqplots with the goodness-of-fit outcomes of the simulated data.
			
		\end{itemize}
		
		\textbf{Answer}
		
		We construct a Pearson chi-squared goodness-of-fit test for
		$$H_0: X\sim \text{Normal} \quad \text{versus} \quad H_A: X \text{ is not Normal}.$$
		Suppose we observe an i.i.d.\ sample $X_1,\dots,X_n$ from the distribution of $X$.
		
		\begin{enumerate}
			\item \textit{Estimate the normal parameters.}
			Under $H_0$, the data should come from some normal distribution $N(\mu,\sigma^2)$, where both $\mu$ and $\sigma^2$ are unknown. We estimate them by
			$$\hat\mu=\bar X=\frac{1}{n}\sum_{i=1}^n X_i, \qquad \hat\sigma^2=\frac{1}{n}\sum_{i=1}^n (X_i-\bar X)^2$$
			(or one may use the unbiased variance estimator; asymptotically this makes no difference for the test).
			
			\item \textit{Partition the real line into $K$ bins.}
			Choose cutpoints $-\infty=a_0<a_1<\cdots<a_{K-1}<a_K=\infty$ and define the bins $B_j=(a_{j-1},a_j]$ for $j=1,\dots,K$. Let the observed count in bin $j$ be
			$$O_j=\sum_{i=1}^n \mathbf{1}_{\{X_i\in B_j\}}, \qquad \text{so that} \qquad \sum_{j=1}^K O_j=n.$$
			A common practical choice is to select the bins so that, under $H_0$, the expected counts are roughly equal; for example, one may take $a_j$ to be the estimated normal quantiles satisfying
			$$\mathbb{P}_{\hat\mu,\hat\sigma}(X\le a_j)=\frac{j}{K},$$
			so that each expected count is about $n/K$.
			
			\item \textit{Compute the expected counts under $H_0$.}
			Under the fitted normal model $N(\hat\mu,\hat\sigma^2)$, the probability of bin $j$ is
			$$p_j=\mathbb{P}_{\hat\mu,\hat\sigma}(X\in B_j)=\Phi\!\left(\frac{a_j-\hat\mu}{\hat\sigma}\right)-\Phi\!\left(\frac{a_{j-1}-\hat\mu}{\hat\sigma}\right),$$
			where $\Phi$ is the standard normal cdf. Hence the expected count in bin $j$ is $E_j=np_j$. For the chi-squared approximation to be reliable, one usually requires all expected counts to be at least about $5$; if some $E_j$ are too small, adjacent bins should be merged.
			
			\item \textit{Form the chi-squared statistic.}
			The Pearson statistic is
			$$Q=\sum_{j=1}^K \frac{(O_j-E_j)^2}{E_j}.$$
			
			\item \textit{Determine the degrees of freedom.}
			If the bin probabilities were fully specified in advance, the statistic would have approximately a $\chi^2_{K-1}$ distribution, since the counts satisfy one constraint $\sum_{j=1}^K O_j=n$. However, since we estimate two parameters $\mu$ and $\sigma^2$ from the data, the degrees of freedom are reduced by $2$. Therefore, under $H_0$,
			$$Q \approx \chi^2_{K-3},$$
			and the test requires $K\ge 4$.
			
			\item \textit{Decision rule.}
			At significance level $\alpha$, reject $H_0$ if $Q>\chi^2_{K-3,\,1-\alpha}$, the $(1-\alpha)$-quantile of the $\chi^2_{K-3}$ distribution. Equivalently, the $p$-value is
			$$p\text{-value}=\mathbb{P}\!\left(\chi^2_{K-3}\ge Q_{\mathrm{obs}}\right),$$
			and we reject when this is less than $\alpha$.
		\end{enumerate}
	
		To compare this test with Q-Q plots, one could simulate:
		\begin{itemize}
			\item Normal data, e.g.\ $X_i\sim N(0,1)$;
			\item Heavy-tailed data, e.g.\ $X_i\sim t_3$;
			\item Skewed data, e.g.\ $X_i\sim \mathrm{Exp}(1)$;
			\item Mixture data, e.g.\ $X_i\sim 0.8\,N(0,1)+0.2\,N(3,1)$.
		\end{itemize}
		For each dataset, draw the Q-Q plot and compute the statistic $Q$. Typically, normal samples produce roughly linear Q-Q plots and non-significant chi-squared tests, while non-normal samples show curvature or tail deviations in the Q-Q plot and tend to produce significant goodness-of-fit statistics.
